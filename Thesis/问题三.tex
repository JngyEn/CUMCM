{6.6.1.1}

b_{i,acceptedRate} = \left(1 - b_{i,1} \cdot (1 - y_{i,1})\right) \cdot \left(1 - c_{i,1} \cdot (1 - x_i)\right) \cdot \left(1 - c_{i,1} \cdot (1 - x_i)\right

b_{i,5} = 1-b_{i,acceptedRate} 

a_{\text{acceptedRate}}  = \left(1 - a_1 \cdot (1 - z_1)\right) \cdot \sum_{i=1}^{3} \left(1 - b_{i,5} \cdot (1 - y_{i,1})\right)

a_{7} =  1 - a_{\text{acceptedRate}}   

{6.6.1.2}

b_{i,N} = \frac{a_N}{1 - b_{i,5} \cdot (1 - y_{i,1})}

c_{i,N} = \frac{b_{i,N}}{1 - c_{i,1} \cdot (1 - x_i)}

a_{\text{detect}} = a_N\cdot a_{7}

a_{\text{good}} = a_N - a_{\text{detect}} 

{6.6.1.3}

\text{purchaseCost}_{c,i} = c_{i,N} \cdot c_{i,2}

\text{purchaseCost}_{c} = \sum_{i=1}^{8}\text{purchaseCost}_{c,i}


{6.6.1.4}

\text{inspectionCost}_{c,i}=x_i\cdot \frac{b_{i,N}}{1 - c_{i,1} \cdot (1 - x_i)}\cdot c_{i,3}

\text{inspectionCost}_{c} = \sum_{i=1}^{8}\text{inspectionCost}_{c,i}

\text{inspectionCost}_{b,i}=y_{i,1}\cdot \frac{a_{N}}{1 - b_{i,1} \cdot (1 - y_{i,1})}\cdot b_{i,3}

\text{inspectionCost}_{b} = \sum_{i=1}^{3}\text{inspectionCost}_{b,i}


\text{inspectionCost}_{a} =  z_1 \cdot a_N \cdot a_3

{6.6.1.5}

\text{assemblyCost }_{b,i} = b_{i,N}\cdot b_{i,2}

\text{totalAssemblyCost }_b =  \sum_{i=1}^{3}\text{assemblyCost }_{b,i}

\text{assemblyCost}_a =  a_N \cdot a_2

{6.6.1.6}

\text{totalIncome} =  a_{\text{good}} \cdot a_5

{6.6.1.7}

\text{productExchangeLoss } = a_{\text{detect}} \cdot a_6


{6.6.1.8}

\text{productDismantlingNum } = z_2 \cdot a_{\text{detect}

\text{halfProductDismantlingNum}_i = k_i \cdot \text{productDismantlingNum }

\text{productDismantlingCost } = \text{productDismantlingNum }\cdot  \sum_{i=1}^{3} (a_2 + a_4 + (1-y_{i,2})\cdot (1 - y_{i,1})\cdot b_{i,5})

\text{halfProductDismantlingCost }_i = \text{halfProductDismantlingNum }_i  \cdot (b_{ i,4} - ( \sum_{k}a_{k,2}) + (\sum_{k}(1 - x_k)\cdot c_{k,3} )

\text{totalDismantlingCost} = \text{productDismantlingCost }  + \sum_{i=1}^{3}  \text{halfProductDismantlingCost }_i

{6.6.1.9}

\text{totalCost} = \text{purchaseCost}_{c} + (\text{inspectionCost}_{a}+\text{inspectionCost}_{b}+\text{inspectionCost}_{c}) +(\text{totalAssemblyCost }_b+\text{assemblyCost}_a) +\text{totalDismantlingCost} + \text{productExchangeLoss } 

\text{totalProfit} = \text{totalIncome}  - \text{totalCost}
